💻 ¿Qué vas a poder hacer?

-- Análisis de ventas por categoría y período SELECT c.nombre_categoria, DATE_FORMAT(v.fecha, '%Y-%m') AS periodo, COUNT(v.id) AS total_ventas, SUM(v.monto) AS ingresos, ROUND(AVG(v.monto), 2) AS ticket_promedio FROM ventas v INNER JOIN categorias c ON v.categoria_id = c.id WHERE v.fecha BETWEEN '2025-01-01' AND '2025-12-31' GROUP BY c.nombre_categoria, periodo HAVING ingresos > 500000 ORDER BY ingresos DESC;
